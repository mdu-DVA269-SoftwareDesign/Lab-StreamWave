\section{User Case Descriptions}
\section*{1: Register Account}

\textbf{Name:} Register Account \\
\textbf{Precondition:} None \\
\textbf{Initiator:} Unauthenticated User \\
\textbf{Goal:} Create a user account to stream music and save playlists

\textbf{Steps to reach the goal:}
\begin{enumerate}
    \item User enters username, email, and password
    \item System checks username is available
    \item System creates account
    \item System displays confirmation message
\end{enumerate}

\textbf{Alternative flows:}
\begin{enumerate}
    \item Email already exists at step 2
    \begin{enumerate}[label=\alph*.]
        \item System shows error message
        \item Resume at 1
    \end{enumerate}
\end{enumerate}

\section*{2: Login}

\textbf{Name:} Login \\
\textbf{Precondition:} User has an account \\
\textbf{Initiator:} Unauthenticated User \\
\textbf{Goal:} Access music collection

\textbf{Steps to reach the goal:}
\begin{enumerate}
    \item User enters username and password
    \item System checks username and password exists and correct
    \item System approves access
    \item User sees music collection on front page
\end{enumerate}

\textbf{Alternative flows:}
\begin{enumerate}
    \item Invalid credentials at step 2
    \begin{enumerate}[label=\alph*.]
        \item System displays error message
        \item Resume at 1
    \end{enumerate}
\end{enumerate}

\newpage

\section*{3: Search and Play Content}

\textbf{Name:} Search and Play Content \\
\textbf{Precondition:} User is logged in \\
\textbf{Initiator:} Registered User \\
\textbf{Goal:} Listen to songs, albums, artists, or podcasts

\textbf{Steps to reach the goal:}
\begin{enumerate}
    \item User enters what to search for
    \item System searches library for that content
    \item System displays search results
    \item User selects content to play
    \item System streams selected content
\end{enumerate}

\textbf{Alternative flows:}
\begin{enumerate}
    \item No results found at step 3
    \begin{enumerate}[label=\alph*.]
        \item System shows message that nothing was found.
        \item Resume at 1
    \end{enumerate}
    \item Free tier user at step 5
    \begin{enumerate}[label=\alph*.]
        \item System plays normal quality audio
        \item Stop
    \end{enumerate}
    \item Premium tier user at step 5
    \begin{enumerate}[label=\alph*.]
        \item System plays high quality audio
        \item Stop
    \end{enumerate}
\end{enumerate}

\section*{4: Manage Playlists}

\textbf{Name:} Manage Playlists \\
\textbf{Precondition:} User is logged in \\
\textbf{Initiator:} Registered User \\
\textbf{Goal:} Organize favourite tracks into personal collections

\textbf{Steps to reach the goal:}
\begin{enumerate}
    \item User selects playlist management option
    \item User chooses action (create, edit, or delete)
    \item For create: User enters name and adds tracks
    \item For edit: User modifies existing playlist
    \item For delete: User confirms deletion
    \item System saves changes
\end{enumerate}

\textbf{Alternative flows:}
\begin{enumerate}
    \item User cancels action at step 3, 4, or 5
    \begin{enumerate}[label=\alph*.]
        \item System cancel changes
        \item Stop
    \end{enumerate}
\end{enumerate}

\newpage

\section*{5: Receive Recommendations}

\textbf{Name:} Receive Recommendations \\
\textbf{Precondition:} User is logged in \\
\textbf{Initiator:} Registered User \\
\textbf{Goal:} Get recommendations based on listening history

\textbf{Steps to reach the goal:}
\begin{enumerate}
    \item User opens front page
    \item System analyzes listening history
    \item System displays similar content
    \item User can stream suggested content
\end{enumerate}

\textbf{Alternative flows:}
\begin{enumerate}
    \item New user with no listening history at step 2
    \begin{enumerate}[label=\alph*.]
        \item System shows trending and popular content
        \item Resume at 4
    \end{enumerate}
\end{enumerate}

\section*{6: Manage Subscription}

\textbf{Name:} Manage Subscription \\
\textbf{Precondition:} User is logged in \\
\textbf{Initiator:} Registered User \\
\textbf{Goal:} Upgrade to premium account for better audio quality

\textbf{Steps to reach the goal:}
\begin{enumerate}
    \item User clicks upgrade account button
    \item User enters payment information
    \item System processes payment
    \item System upgrades account to premium
    \item System confirms upgrade
\end{enumerate}

\textbf{Alternative flows:}
\begin{enumerate}
    \item Payment fails at step 3
    \begin{enumerate}[label=\alph*.]
        \item System displays payment error
        \item Resume at 2
    \end{enumerate}
    \item User cancels at step 3
    \begin{enumerate}[label=\alph*.]
        \item System returns to previous page
        \item Stop
    \end{enumerate}
\end{enumerate}

\section*{7: Upload and Manage Catalog}

\textbf{Name:} Upload and Manage Catalog \\
\textbf{Precondition:} Artist is logged in \\
\textbf{Initiator:} Artist \\
\textbf{Goal:} Upload music and manage album collections with cover images

\textbf{Steps to reach the goal:}
\begin{enumerate}
    \item Artist chooses to upload song or create album
    \item Artist uploads audio files
    \item Artist enters title, genre, album name
    \item Artist uploads cover image
    \item System processes and publishes content
    \item System confirms if upload was successful
\end{enumerate}

\textbf{Alternative flows:}
\begin{enumerate}
    \item Invalid file format at step 2
    \begin{enumerate}[label=\alph*.]
        \item System displays error message
        \item Resume at 2
    \end{enumerate}
    \item Edit existing content at step 2
    \begin{enumerate}[label=\alph*.]
        \item Artist selects existing content
        \item Artist modifies details or replaces files
        \item System updates content
        \item Resume at 6
    \end{enumerate}
    \item Delete content at step 2
    \begin{enumerate}[label=\alph*.]
        \item Artist selects content to delete
        \item System confirms deletion
        \item System removes content
        \item Stop
    \end{enumerate}
\end{enumerate}

\section*{8: Manage Users}

\textbf{Name:} Manage Users \\
\textbf{Precondition:} Administrator is logged in \\
\textbf{Initiator:} Administrator \\
\textbf{Goal:} Add, edit, or delete user accounts for system security

\textbf{Steps to reach the goal:}
\begin{enumerate}
    \item Administrator opens user management
    \item Administrator selects action (add, edit, or delete)
    \item For add: Administrator enters new user details
    \item For edit: Administrator modifies user information
    \item For delete: Administrator confirms deletion
    \item System executes action
    \item System confirms changes
\end{enumerate}

\textbf{Alternative flows:}
\begin{enumerate}
    \item Username already exists at step 3
    \begin{enumerate}[label=\alph*.]
        \item System shows error message
        \item Resume at 3
    \end{enumerate}
    \item User not found at step 4 or 5
    \begin{enumerate}[label=\alph*.]
        \item System displays "User not found"
        \item Resume at 2
    \end{enumerate}
\end{enumerate}

\newpage

\section*{9: Manage Content and Tiers}

\textbf{Name:} Manage Content and Tiers \\
\textbf{Precondition:} Administrator is logged in \\
\textbf{Initiator:} Administrator \\
\textbf{Goal:} Manage content items and handle audio quality for Free/Premium tiers

\textbf{Steps to reach the goal:}
\begin{enumerate}
    \item Administrator opens content and tier management
    \item Administrator selects management option (content or tiers)
    \item For content: Administrator adds, edits, or deletes content items
    \item For tiers: Administrator sets audio quality levels for Free and Premium
    \item System applies changes
    \item System confirms update
\end{enumerate}

\textbf{Alternative flows:}
\begin{enumerate}
    \item Content upload fails at step 3
    \begin{enumerate}[label=\alph*.]
        \item System displays error message
        \item Resume at 3
    \end{enumerate}
    \item Invalid quality settings at step 4
    \begin{enumerate}[label=\alph*.]
        \item System shows validation error
        \item Resume at 4
    \end{enumerate}
\end{enumerate}
